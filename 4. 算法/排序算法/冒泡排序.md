# 冒泡排序
```
//  冒泡排序
function bubble(array){
  checkArray(array)
  // if(!Array.isArray(array)) {
  //   return "not a array"
  // }
  for(let i = array.length - 1 ; i>0 ; i--){
    for(let j =0 ; j<i ; j++){
      if(array[j]>array[j+1]){
        swap(array,j,j+1)
      }
    }
  }
  return array
}
```
