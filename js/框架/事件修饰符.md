# Vue 指令之 v-on的缩写 和 事件修饰符
```
<div id="app">
    <div class="inner" @click='divHandler'>
        <!-- 使用 .stop 阻止冒泡 -->
        <input type="button" value="戳他，阻止冒泡" @click.stop="btnHandler">
    </div>
    <!-- .prevent 阻止默认事件 -->
    <a href="http://www.baidu.com" @click.prevent="linkClick">百度</a>

    <div class="inner" @click.capture='divHandler'>
        <!-- 使用 .capture 捕获 -->
        <input type="button" value="戳他，捕获" @click="btnHandler">
    </div>

    <!-- 演示 .stop 和 .self 的区别 -->
    <!-- .self 只会阻止自己身上冒泡行为的触发，并不会阻止 -->
</div>
<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
<script>
    var vm = new Vue({
        el:"#app",
        methods:{
            divHandler(){
                console.log("我是div")
            },
            btnHandler(){
                console.log("我是btn")
            },
            linkClick(){
                console.log("触发链接的点击事件")
            }
        }
    })
</script>
```
